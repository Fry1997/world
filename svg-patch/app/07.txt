  function selectCountry(code) {
    const country = countryByCode.get(code), state = currentState();
    if (!country) return;
    selectedCode = code;
    const guess = (state.guesses || []).find(item => item.code === code);
    elements.selectionName.textContent = country.name;
    elements.selectionMeta.textContent = guess ? `Already guessed · ${formatDistance(guess.distance)} away` : state.complete ? "This game is complete" : "Use this as your next guess";
    elements.selectionUse.disabled = Boolean(guess || state.complete);
    elements.selection.classList.remove("is-hidden"); refreshMap();
  }

  function clearSelection() {
    selectedCode = null; elements.selection.classList.add("is-hidden"); refreshMap();
  }

  function useSelection() {
    if (!selectedCode) return;
    const country = countryByCode.get(selectedCode), state = currentState();
    if (!country || state.complete || (state.guesses || []).some(item => item.code === selectedCode)) return;
    countryInput.value = country.name;
    countryInput.dispatchEvent(new Event("input", { bubbles: true }));
    clearSelection();
    countryInput.scrollIntoView({ behavior: "smooth", block: "center" }); countryInput.focus();
  }
