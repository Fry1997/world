x];
      const next = index === xs.length - 1 ? xs[0] + 1 : xs[index + 1];
      if (next - current > largestGap) { largestGap = next - current; gapIndex = index; }
    }
    const minX = gapIndex === xs.length - 1 ? xs[0] : xs[gapIndex + 1];
    const maxX = gapIndex === xs.length - 1 ? xs[xs.length - 1] : xs[gapIndex] + 1;
    return { minX, maxX, minY, maxY, centerX: ((minX + maxX) / 2) % 1, centerY: (minY + maxY) / 2 };
  }

  function buildGraticule() {
    const fragment = document.createDocumentFragment();
    for (let longitude = -150; longitude <= 150; longitude += 30) {
      const points = [];
      for (let latitude = -80; latitude <= 80; latitude += 4) points.push([longitude, latitude]);
      const path = document.createElementNS(SVG_NS, "path");
      path.setAttribute("d", ringToPath(points).replace(/ Z$/, ""));
      path.setAttribute("class", longitude === 0 ? "map-grid-line is-major" : "map-grid-line");
      path.setAttribute("vector-effect", "non-scaling-stroke"); fragment.appendChild(path);
    }
    for (let latitude = -60; latitude <= 60; latitude += 30) {
      const points = [];
      for (let longitude = -180; longitude <= 180; longitude += 5) points.push([longitude, latitude]);
      const path = document.createElementNS(SVG_NS, "path");
      path.setAttribute("d", ringToPath(points).replace(/ Z$/, ""));
      path.setAttribute("class", latitude === 0 ? "map-grid-line is-major" : "map-grid-line");
      path.setAttribute("vector-effect", "non-scaling-stroke"); fragment.appendChild(path);
    }
    elements.graticule.appendChild(fragment);
  }
