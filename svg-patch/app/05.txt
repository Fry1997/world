  function fitZoom() {
    const rect = elements.world.getBoundingClientRect();
    if (!rect.width || !rect.height) return 0.5;
    return Math.max(0.25, Math.min(2, Math.log2(Math.min(rect.width / 256, rect.height / (256 * 0.78)))));
  }

  function clampCamera() {
    const rect = elements.world.getBoundingClientRect();
    if (!rect.width || !rect.height) return;
    mapCamera.zoom = Math.max(0.25, Math.min(7, mapCamera.zoom));
    const worldSize = 256 * Math.pow(2, mapCamera.zoom);
    const halfX = rect.width / (2 * worldSize), halfY = rect.height / (2 * worldSize);
    mapCamera.x = halfX >= 0.5 ? 0.5 : Math.max(halfX, Math.min(1 - halfX, mapCamera.x));
    mapCamera.y = halfY >= 0.5 ? 0.5 : Math.max(halfY, Math.min(1 - halfY, mapCamera.y));
  }

  function scheduleRender() {
    if (renderFrame) return;
    renderFrame = requestAnimationFrame(() => { renderFrame = 0; renderScene(); });
  }

  function screenX(worldX, worldSize, width) {
    const wrapped = normaliseWrappedX(worldX, mapCamera.x);
    return width / 2 + (wrapped - mapCamera.x) * worldSize;
  }
